  <script>
    (function(){
      const track = document.getElementById('carouselTrack');
      if (!track) return;

      const container = track.parentElement;
      let raf;
      let lastTime = null;
      let speed = 0.15; // px per ms ‚Üí ajuste de velocidade (‚Üë para mais r√°pido)
      let isPointerDown = false;
      let startX = 0;
      let scrollStart = 0;
      let resumeTimeout;
      const RESUME_DELAY = 1800; // ms at√© voltar ao auto ap√≥s intera√ß√£o

      // ensure track is horizontally scrollable
      container.style.overflowX = 'hidden';
      track.style.willChange = 'transform';

      // auto-scroll loop using scrollLeft manipulation
      function step(ts){
        if (!lastTime) lastTime = ts;
        const dt = ts - lastTime;
        lastTime = ts;

        if (!isPointerDown) {
          container.scrollLeft += speed * dt;
          // loop effect: when scrolled to near end, rewind smoothly to start
          if (container.scrollLeft >= track.scrollWidth / 2) {
            // jump back but keep smooth illusion
            container.scrollLeft = container.scrollLeft - (track.scrollWidth / 2);
          }
        }
        raf = requestAnimationFrame(step);
      }
      raf = requestAnimationFrame(step);

      // Pointer drag handlers
      function onPointerDown(e){
        isPointerDown = true;
        container.classList.add('is-dragging');
        startX = (e.touches ? e.touches[0].pageX : e.pageX) || e.pageX;
        scrollStart = container.scrollLeft;
        // stop auto-resume while dragging
        if (resumeTimeout) clearTimeout(resumeTimeout);
      }
      function onPointerMove(e){
        if(!isPointerDown) return;
        const x = (e.touches ? e.touches[0].pageX : e.pageX) || e.pageX;
        const dx = x - startX;
        container.scrollLeft = scrollStart - dx;
      }
      function onPointerUp(e){
        isPointerDown = false;
        container.classList.remove('is-dragging');
        // resume after short delay
        if (resumeTimeout) clearTimeout(resumeTimeout);
        resumeTimeout = setTimeout(()=> { lastTime = null; }, RESUME_DELAY);
      }

      // events
      track.addEventListener('mousedown', onPointerDown);
      track.addEventListener('touchstart', onPointerDown, {passive:true});
      window.addEventListener('mousemove', onPointerMove);
      window.addEventListener('touchmove', onPointerMove, {passive:true});
      window.addEventListener('mouseup', onPointerUp);
      window.addEventListener('touchend', onPointerUp);

      // pause auto on wheel, resume after delay
      let wheelTO;
      container.addEventListener('wheel', (ev) => {
        // user controlling: pause auto (by setting lastTime null temporarily)
        if (wheelTO) clearTimeout(wheelTO);
        lastTime = null;
        wheelTO = setTimeout(()=> { lastTime = null; }, RESUME_DELAY);
      }, {passive:true});

      // accessibility: pause when tab not visible
      document.addEventListener('visibilitychange', () => {
        if (document.hidden) cancelAnimationFrame(raf);
        else raf = requestAnimationFrame(step);
      });
    })();
